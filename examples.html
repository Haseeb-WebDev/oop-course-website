Java Examples

Java is a pure object-oriented language that strongly enforces OOP principles. Here are comprehensive examples showing all four OOP concepts. 
public class Student {
    // Private fields (encapsulated)
    private String name;
    private int age;
    private double gpa;
    private String studentId;
    
    // Constructor
    public Student(String name, int age, String studentId) {
        this.name = name;
        this.age = age;
        this.studentId = studentId;
        this.gpa = 0.0;
    }
    
    // Getter methods
    public String getName() {
        return name;
    }
    
    public int getAge() {
        return age;
    }
    
    public double getGpa() {
        return gpa;
    }
    
    public String getStudentId() {
        return studentId;
    }
    
    // Setter methods with validation
    public void setGpa(double gpa) {
        if (gpa >= 0.0 && gpa <= 4.0) {
            this.gpa = gpa;
        } else {
            System.out.println("Invalid GPA!");
        }
    }
    
    public void setName(String name) {
        if (name != null && !name.trim().isEmpty()) {
            this.name = name;
        }
    }
    
    // Display student information
    public void displayInfo() {
        System.out.println("Student: " + name);
        System.out.println("ID: " + studentId);
        System.out.println("Age: " + age);
        System.out.println("GPA: " + gpa);
    }
}
// Base class
abstract class Vehicle {
    protected String brand;
    protected String model;
    protected int year;
    protected double price;
    
    public Vehicle(String brand, String model, int year, double price) {
        this.brand = brand;
        this.model = model;
        this.year = year;
        this.price = price;
    }
    
    // Abstract method
    public abstract void start();
    
    // Concrete methods
    public void displayInfo() {
        System.out.println(year + " " + brand + " " + model);
        System.out.println("Price: $" + price);
    }
    
    public void stop() {
        System.out.println("Vehicle stopped.");
    }
}

// Derived class
class Car extends Vehicle {
    private int doors;
    private String fuelType;
    
    public Car(String brand, String model, int year, 
               double price, int doors, String fuelType) {
        super(brand, model, year, price);
        this.doors = doors;
        this.fuelType = fuelType;
    }
    
    @Override
    public void start() {
        System.out.println("Car engine started with key.");
    }
    
    public void honk() {
        System.out.println("Car honks: Beep! Beep!");
    }
    
    @Override
    public void displayInfo() {
        super.displayInfo();
        System.out.println("Doors: " + doors);
        System.out.println("Fuel Type: " + fuelType);
    }
}

Python Examples

Python supports OOP with a clean and intuitive syntax. Here are examples demonstrating OOP concepts in Python. 
from abc import ABC, abstractmethod

class BankAccount(ABC):
    def __init__(self, account_number, owner_name):
        self._account_number = account_number  # Protected
        self._owner_name = owner_name
        self.__balance = 0.0  # Private
        self.__transaction_history = []  # Private
    
    # Getter methods
    @property
    def balance(self):
        return self.__balance
    
    @property
    def account_number(self):
        return self._account_number
    
    @property
    def owner_name(self):
        return self._owner_name
    
    # Protected method
    def _validate_amount(self, amount):
        return amount > 0
    
    # Private method
    def __add_transaction(self, transaction_type, amount):
        self.__transaction_history.append({
            'type': transaction_type,
            'amount': amount,
            'balance': self.__balance
        })
    
    # Abstract method
    @abstractmethod
    def calculate_interest(self):
        pass
    
    # Public methods
    def deposit(self, amount):
        if self._validate_amount(amount):
            self.__balance += amount
            self.__add_transaction('deposit', amount)
            return True
        return False
    
    def withdraw(self, amount):
        if self._validate_amount(amount) and amount <= self.__balance:
            self.__balance -= amount
            self.__add_transaction('withdrawal', amount)
            return True
        return False
    
    def get_statement(self):
        return self.__transaction_history.copy()

# Concrete implementation
class SavingsAccount(BankAccount):
    def __init__(self, account_number, owner_name, interest_rate):
        super().__init__(account_number, owner_name)
        self.interest_rate = interest_rate
    
    def calculate_interest(self):
        return self.balance * self.interest_rate / 100
    
    def add_interest(self):
        interest = self.calculate_interest()
        if interest > 0:
            self.deposit(interest)
            print(f"Interest added: ${interest:.2f}")

class CheckingAccount(BankAccount):
    def __init__(self, account_number, owner_name, overdraft_limit):
        super().__init__(account_number, owner_name)
        self.overdraft_limit = overdraft_limit
    
    def calculate_interest(self):
        return 0  # No interest for checking accounts

class Animal:
    def __init__(self, name, species):
        self.name = name
        self.species = species
    
    def make_sound(self):
        pass
    
    def move(self):
        print(f"{self.name} is moving")
    
    def __str__(self):
        return f"{self.name} the {self.species}"

class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name, "Dog")
        self.breed = breed
    
    def make_sound(self):
        return "Woof! Woof!"
    
    def move(self):
        print(f"{self.name} runs on four legs")
    
    def fetch(self):
        return f"{self.name} fetches the ball"

class Cat(Animal):
    def __init__(self, name, breed):
        super().__init__(name, "Cat")
        self.breed = breed
    
    def make_sound(self):
        return "Meow! Meow!"
    
    def move(self):
        print(f"{self.name} prowls silently")
    
    def climb(self):
        return f"{self.name} climbs the tree"

class Bird(Animal):
    def __init__(self, name, species, can_fly=True):
        super().__init__(name, species)
        self.can_fly = can_fly
    
    def make_sound(self):
        return "Chirp! Chirp!"
    
    def move(self):
        if self.can_fly:
            print(f"{self.name} flies through the air")
        else:
            print(f"{self.name} walks on the ground")

# Polymorphism demonstration
def animal_sounds(animals):
    for animal in animals:
        print(f"{animal} says: {animal.make_sound()}")
        animal.move()
        print()

JavaScript Examples

JavaScript ES6+ introduced class syntax that makes OOP more intuitive. Here are modern JavaScript OOP examples. 
class Employee {
    // Private fields (ES2022)
    #salary;
    #ssn;
    #employeeId;
    
    constructor(name, department, salary, ssn) {
        this.name = name;
        this.department = department;
        this.#salary = salary;
        this.#ssn = ssn;
        this.#employeeId = this.#generateEmployeeId();
        this.projects = [];
    }
    
    // Private method
    #generateEmployeeId() {
        return 'EMP' + Date.now().toString().slice(-6);
    }
    
    // Getter methods
    get employeeId() {
        return this.#employeeId;
    }
    
    get salary() {
        return this.#salary;
    }
    
    // Setter with validation
    set salary(newSalary) {
        if (newSalary > 0) {
            this.#salary = newSalary;
        } else {
            throw new Error('Salary must be positive');
        }
    }
    
    // Public methods
    addProject(project) {
        this.projects.push(project);
    }
    
    getInfo() {
        return {
            id: this.#employeeId,
            name: this.name,
            department: this.department,
            projectCount: this.projects.length
        };
    }
    
    // Method to safely display salary (only last 2 digits)
    getObfuscatedSalary() {
        const salaryStr = this.#salary.toString();
        return '*'.repeat(salaryStr.length - 2) + salaryStr.slice(-2);
    }
    
    calculateBonus() {
        return this.#salary * 0.1; // 10% bonus
    }
    
    toString() {
        return `Employee: ${this.name} (${this.#employeeId}) - ${this.department}`;
    }
}

// Base class
class Shape {
    constructor(color) {
        this.color = color;
        this.created = new Date();
    }
    
    // Method to be overridden
    calculateArea() {
        throw new Error("calculateArea method must be implemented");
    }
    
    calculatePerimeter() {
        throw new Error("calculatePerimeter method must be implemented");
    }
    
    getInfo() {
        return {
            type: this.constructor.name,
            color: this.color,
            area: this.calculateArea(),
            perimeter: this.calculatePerimeter(),
            created: this.created
        };
    }
}

// Derived classes
class Circle extends Shape {
    constructor(color, radius) {
        super(color);
        this.radius = radius;
    }
    
    calculateArea() {
        return Math.PI * this.radius ** 2;
    }
    
    calculatePerimeter() {
        return 2 * Math.PI * this.radius;
    }
    
    // Additional method specific to Circle
    getDiameter() {
        return this.radius * 2;
    }
}

class Rectangle extends Shape {
    constructor(color, width, height) {
        super(color);
        this.width = width;
        this.height = height;
    }
    
    calculateArea() {
        return this.width * this.height;
    }
    
    calculatePerimeter() {
        return 2 * (this.width + this.height);
    }
    
    // Additional method
    isSquare() {
        return this.width === this.height;
    }
}

class Triangle extends Shape {
    constructor(color, base, height, sideA, sideB) {
        super(color);
        this.base = base;
        this.height = height;
        this.sideA = sideA;
        this.sideB = sideB;
    }
    
    calculateArea() {
        return (this.base * this.height) / 2;
    }
    
    calculatePerimeter() {
        return this.base + this.sideA + this.sideB;
    }
}

// Polymorphism in action
function displayShapesInfo(shapes) {
    shapes.forEach(shape => {
        console.log(`${shape.constructor.name}:`);
        console.log(`  Color: ${shape.color}`);
        console.log(`  Area: ${shape.calculateArea().toFixed(2)}`);
        console.log(`  Perimeter: ${shape.calculatePerimeter().toFixed(2)}`);
        console.log('---');
    });
}

// Usage
const shapes = [
    new Circle("red", 5),
    new Rectangle("blue", 4, 6),
    new Triangle("green", 3, 4, 5, 5)
];

displayShapesInfo(shapes);
